Stack 구현

<!--truncate-->

```h title="stack.h"
#ifndef STACK_H
#define STACK_H

#include <cstddef>

class Stack
{
private:
    struct Node
    {
        int data;
        Node *next;
    };

    Node *top;
    size_t count;

public:
    Stack();
    ~Stack();
    bool empty();
    size_t size();
    void push(int value);
    void pop();
    int pick();
};

#endif // STACK_H
```

```cpp title="stack.cpp"
#include "stack.h"

// Constructor
Stack::Stack()
{
    top = nullptr;
    count = 0;
}

// Destructor
Stack::~Stack()
{
    while (!empty())
    {
        pop();
    }
}

// Check if the stack is empty
bool Stack::empty()
{
    return (top == nullptr);
}

// Return the size of the stack
size_t Stack::size()
{
    return count;
}

// Add an element to the top of the stack
void Stack::push(int value)
{
    Node *newNode = new Node;
    newNode->data = value;
    newNode->next = top;
    top = newNode;
    count++;
}

// Remove the element from the top of the stack
void Stack::pop()
{
    if (!empty())
    {
        Node *temp = top;
        top = top->next;
        delete temp;
        count--;
    }
}

// Get the value of the top element in the stack and return -1 if the stack is empty
int Stack::pick()
{
    if (!empty())
    {
        return top->data;
    }
    else
    {
        return -1; // Return an error value
    }
}
```
